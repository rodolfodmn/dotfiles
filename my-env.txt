from pynput.keyboard import Key, Controller
import time

keyboard = Controller()

time.sleep(4)
atth = 3

def att():
    keyboard.press('7')
    time.sleep(1)
    keyboard.release('7')

def attheal():
    for x in range(atth):
        keyboard.press('7')
        keyboard.press('8')
        time.sleep(1)
        keyboard.release('7')
        keyboard.release('8')
        time.sleep(1)

def mup(t):
    keyboard.press('e')
    for x in range(t):
        time.sleep(1)
    keyboard.release('e')

def mdown(t):
    keyboard.press('d')
    time.sleep(t)
    keyboard.release('d')

def mlef(t):
    keyboard.press('s')
    for x in range(t):
        keyboard.press('7')
        time.sleep(1)
        keyboard.release('7')

    keyboard.release('s')

def mrig(t):
    keyboard.press('r')
    for x in range(t):
        keyboard.press('7')
        time.sleep(1)
        keyboard.release('7')

    keyboard.release('r')


def vai():
    attheal()
    mrig(4)
    attheal()
    mrig(3)
    attheal()
    mrig(3)
    attheal()
    mrig(3)
    attheal()
    mrig(3)
    attheal()

def volta():
    mlef(16)

for x in range(5):

    vai()
